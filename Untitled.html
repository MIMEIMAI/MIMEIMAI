            text-align: center;

        }

        #time {

            font-size: 48px;

            margin-bottom: 20px;

        }

        button {

            font-size: 20px;

            padding: 10px 20px;

            margin: 5px;

        }

<body>

    <div id="pomodoro">

        <div id="time">25:00</div>

        <button onclick="startTimer()">Start</button>

        <button onclick="resetTimer()">Reset</button>

    </div>

    <script>

        let timer;

        let isRunning = false;

        let minutes = 25;

        let seconds = 0;


        function updateDisplay() {

            const minutesDisplay = minutes < 10 ? `0${minutes}` : minutes;

            const secondsDisplay = seconds < 10 ? `0${seconds}` : seconds;

            document.getElementById('time').textContent = `${minutesDisplay}:${secondsDisplay}`;

        }


        function startTimer() {

            if (isRunning) return;

            isRunning = true;

            timer = setInterval(() => {

                if (seconds === 0) {

                    if (minutes === 0) {

                        clearInterval(timer);

                        isRunning = false;

                        alert('Time to take a break!');

                    } else {

                        minutes--;

                        seconds = 59;

                    }

                } else {

                    seconds--;

                }

                updateDisplay();

            }, 1000);

        }


        function resetTimer() {

            clearInterval(timer);

            isRunning = false;

            minutes = 25;

            seconds = 0;

            updateDisplay();

        }


        updateDisplay();

    </script>

</body>
